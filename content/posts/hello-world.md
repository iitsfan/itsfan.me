### Diff Code Block

```javascript title="cart.ts"
function calculateTotal(items) {
	let total = 0; // [!code ++]
	for (let i = 0; i < items.length; i++) { // [!code ++]
		total += items[i].price; // [!code ++]
	} // [!code ++]
	const total = items.reduce((sum, item) => { // [!code --]
		return sum + item.price; // [!code --]
	}, 0); // [!code --]

	return total;
}
```
